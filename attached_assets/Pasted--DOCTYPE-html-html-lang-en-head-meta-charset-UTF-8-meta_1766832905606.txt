        const App = {
            quizData: null,
            currentQuestionIndex: 0,
            userAnswers: [],
            quizHistory: [],
            currentQuizId: null,
            navigationDirection: 'forward',
            optionLabels: ['A', 'B', 'C', 'D', 'E', 'F', 'G', 'H'],
            STORAGE_KEY: 'quiz_history',

            async saveQuizFile(quizData) {
                try {
                    const quizId = 'quiz_' + Date.now();
                    await window.storage.set(quizId, JSON.stringify(quizData));
                    console.log('Quiz saved:', quizId);
                    return quizId;
                } catch (error) {
                    console.error('Error saving quiz:', error);
                    return null;
                }
            },

            async loadQuizFile(quizId) {
                try {
                    const result = await window.storage.get(quizId);
                    if (result && result.value) {
                        return JSON.parse(result.value);
                    }
                    return null;
                } catch (error) {
                    console.error('Error loading quiz:', error);
                    return null;
                }
            },

            async loadQuizHistory() {
                try {
                    const result = await window.storage.get(this.STORAGE_KEY);
                    if (result && result.value) {
                        this.quizHistory = JSON.parse(result.value);
                    } else {
                        this.quizHistory = [];
                    }
                } catch (error) {
                    this.quizHistory = [];
                }
                this.displayQuizHistory();
            },

            async saveQuizHistory() {
                try {
                    await window.storage.set(this.STORAGE_KEY, JSON.stringify(this.quizHistory));
                } catch (error) {
                    console.error('Error saving history:', error);
                }
            },

            async saveQuizResult(title, score, total, quizId) {
                const result = {
                    title,
                    score,
                    total,
                    date: new Date().toLocaleDateString(),
                    time: new Date().toLocaleTimeString(),
                    percentage: Math.round((score / total) * 100),
                    timestamp: Date.now(),
                    quizId
                };
                
                this.quizHistory.unshift(result);
                if (this.quizHistory.length > 10) this.quizHistory.pop();
                
                await this.saveQuizHistory();
                this.displayQuizHistory();
            },

            displayQuizHistory() {
                const container = document.getElementById('quizHistoryContainer');
                const section = document.getElementById('recentQuizzesSection');
                
                if (this.quizHistory.length === 0) {
                    section.style.display = 'none';
                    return;
                }
                
                section.style.display = 'block';
                container.innerHTML = this.quizHistory.map((quiz) => `
                    <div class="quiz-history-item">
                        <div>
                            <div class="quiz-history-title">${quiz.title}</div>
                            <div class="quiz-history-date">${quiz.date} at ${quiz.time}</div>
                        </div>
                        <div style="display: flex; align-items: center; gap: 15px;">
                            <div class="quiz-history-score">${quiz.score}/${quiz.total} (${quiz.percentage}%)</div>
                            <button class="btn btn-primary" style="padding: 8px 16px; font-size: 0.85em;" onclick="App.reattemptQuiz('${quiz.quizId}')">
                                üîÑ Reattempt
                            </button>
                        </div>
                    </div>
                `).join('');
            },

            async handleFile(file) {
                const reader = new FileReader();
                
                reader.onload = async (e) => {
                    try {

            sanitizeHTML(text) {
                // Allow basic HTML tags but sanitize to prevent XSS
                const div = document.createElement('div');
                div.textContent = text;
                let sanitized = div.innerHTML;
                
                // Allow certain HTML tags
                const allowedTags = ['b', 'i', 'em', 'strong', 'u', 'br', 'p', 'span', 'sub', 'sup', 'code', 'pre'];
                const tagPattern = new RegExp(`&lt;(/?)(${allowedTags.join('|')})(\\s[^&]*?)?&gt;`, 'gi');
                
                sanitized = sanitized.replace(tagPattern, '<$1$2$3>');
                
                return sanitized;
            },

            showResults() {
                let correctCount = 0;
                let answeredCount = 0;
                const incorrectQuestions = [];
                const topicStats = {};
                
                this.quizData.data.forEach((question, index) => {
                    const topic = this.questionTopics && this.questionTopics[index] 
                        ? this.questionTopics[index] 
                        : this.extractTopicFallback(question.question);
                    
                    if (!topicStats[topic]) {
                        topicStats[topic] = { correct: 0, total: 0 };
                    }
                    
                    if (this.userAnswers[index] !== null) {
                        answeredCount++;
                        topicStats[topic].total++;
                        
                        if (this.userAnswers[index] === question.correct_answer) {
                            correctCount++;
                            topicStats[topic].correct++;
                        } else {
                            incorrectQuestions.push({
                                question: question.question,
                                userAnswer: this.userAnswers[index],
                                correctAnswer: question.correct_answer,
                                topic: topic
                            });
                        }
                    }
                });
                
                const total = this.quizData.data.length;
                const incorrectCount = answeredCount - correctCount;
                const percentage = answeredCount > 0 ? Math.round((correctCount / answeredCount) * 100) : 0;
                
                document.getElementById('quizSection').style.display = 'none';
                document.getElementById('resultsSection').style.display = 'block';
                
                document.getElementById('scoreDisplay').textContent = percentage + '%';
                document.getElementById('finalCorrectCount').textContent = correctCount;
                document.getElementById('finalIncorrectCount').textContent = incorrectCount;
                document.getElementById('totalCount').textContent = answeredCount + '/' + total;
                
                // Display topics summary
                const topicsSummaryEl = document.getElementById('topicsSummary');
                if (Object.keys(topicStats).length > 0) {
                    // Filter out topics with 0 questions
                    const sortedTopics = Object.entries(topicStats)
                        .filter(([topic, stats]) => stats.total > 0)
                        .sort((a, b) => b[1].total - a[1].total);
